Supabase RLS - מדריך שלב-אחר-שלב

✅ מה כבר עשית

יצרת JWT Template ב-Clerk עם HS256 + Supabase JWT Secret. מצוין! זה אומר ש-Supabase כבר יודע לאמת את הטוקנים שלך.

💡 למה זה עובד?

Supabase משתמש באותו JWT Secret גם לחתום על טוקנים וגם לאמת אותם. כשהגדרת ב-Clerk את ה-JWT Secret של Supabase עם HS256, כל טוקן ש-Clerk מייצר מאומת אוטומטית ע"י Supabase - בלי שום הגדרה נוספת בצד של Supabase!

1

עדכון Supabase Client לשימוש בטוקן Clerk

צריך ליצור פונקציה שמחזירה Supabase client עם הטוקן של Clerk:

// src/api/supabaseClient.js

import { createClient } from '@supabase/supabase-js';

const supabaseUrl = import.meta.env.VITE_SUPABASE_URL;
const supabaseAnonKey = import.meta.env.VITE_SUPABASE_ANON_KEY;

// Client ציבורי (ללא auth) - לקריאות ציבוריות
export const supabase = createClient(supabaseUrl, supabaseAnonKey);

// Client מאומת - עם טוקן Clerk
// קורא לזה מתוך קומפוננטות React (כי צריך hook של Clerk)
export function createAuthenticatedClient(supabaseToken) {
  return createClient(supabaseUrl, supabaseAnonKey, {
    global: {
      headers: {
        Authorization: `Bearer ${supabaseToken}`,
      },
    },
  });
}
2

יצירת Hook לשימוש נוח

Hook שנותן Supabase client מאומת בכל קומפוננטה:

// src/hooks/useSupabaseAuth.js

import { useAuth } from '@clerk/clerk-react';
import { useMemo } from 'react';
import { createAuthenticatedClient } from '../api/supabaseClient';

export function useSupabaseAuth() {
  const { getToken, isSignedIn } = useAuth();

  // פונקציה שמחזירה client מאומת
  const getAuthClient = async () => {
    const token = await getToken({ template: 'supabase' });
    if (!token) {
      throw new Error('No Supabase token available');
    }
    return createAuthenticatedClient(token);
  };

  return { getAuthClient, isSignedIn };
}

בדיקה שהטוקן עובד

הוסף בדיקה מהירה בקונסולה או בקומפוננטת טסט:

// קומפוננטת בדיקה זמנית - src/components/TestRLS.jsx

import { useSupabaseAuth } from '../hooks/useSupabaseAuth';

export default function TestRLS() {
  const { getAuthClient } = useSupabaseAuth();

  const testConnection = async () => {
    try {
      const client = await getAuthClient();

      // בדיקה 1: האם הטוקן תקין
      const { data: { user }, error } = await client.auth.getUser();
      console.log('Supabase user from Clerk token:', user);
      console.log('Error:', error);

      // בדיקה 2: נסה לקרוא מטבלה
      const { data, error: queryError } = await client
        .from('books')
        .select('id, title')
        .limit(5);
      console.log('Books query result:', data);
      console.log('Query error:', queryError);

      alert(user
        ? `Token works! User sub: ${user.id}`
        : `Token issue: ${error?.message}`
      );
    } catch (err) {
      console.error('Test failed:', err);
      alert('Failed: ' + err.message);
    }
  };

  return (
    <button onClick={testConnection}
      style={{ padding: '12px 24px', background: '#7C3AED',
               color: 'white', borderRadius: '8px', border: 'none',
               cursor: 'pointer', fontSize: '16px' }}>
      Test RLS Connection
    </button>
  );
}

⚠ חשוב!

ברגע שמפעילים RLS על טבלה, אף אחד לא יכול לקרוא/לכתוב אליה עד שיוצרים policies. לכן - תמיד תפעיל RLS + תיצור policies באותו ה-SQL query.

5

יצירת Policies (מדיניות גישה)

ה-JWT מ-Clerk מכיל sub (Clerk user ID). נשתמש בו ב-policies:

-- ============================================
-- שלב 5: יצירת Policies
-- ============================================

-- פונקציית עזר: שליפת user_id מהטוקן
CREATE OR REPLACE FUNCTION auth.clerk_user_id()
RETURNS TEXT AS $$
  SELECT nullif(
    current_setting('request.jwt.claims', true)::json->>'sub',
    ''
  );
$$ LANGUAGE sql STABLE;

-- ============================================
-- books: כל אחד קורא ספרים ציבוריים, רק הבעלים עורך
-- ============================================
CREATE POLICY "Anyone can read public books"
  ON books FOR SELECT
  USING (is_public = true OR user_id = auth.clerk_user_id());

CREATE POLICY "Users can insert own books"
  ON books FOR INSERT
  WITH CHECK (user_id = auth.clerk_user_id());

CREATE POLICY "Users can update own books"
  ON books FOR UPDATE
  USING (user_id = auth.clerk_user_id())
  WITH CHECK (user_id = auth.clerk_user_id());

CREATE POLICY "Users can delete own books"
  ON books FOR DELETE
  USING (user_id = auth.clerk_user_id());

-- ============================================
-- characters: רק הבעלים רואה ועורך
-- ============================================
CREATE POLICY "Users can read own characters"
  ON characters FOR SELECT
  USING (user_id = auth.clerk_user_id());

CREATE POLICY "Users can insert own characters"
  ON characters FOR INSERT
  WITH CHECK (user_id = auth.clerk_user_id());

CREATE POLICY "Users can update own characters"
  ON characters FOR UPDATE
  USING (user_id = auth.clerk_user_id());

CREATE POLICY "Users can delete own characters"
  ON characters FOR DELETE
  USING (user_id = auth.clerk_user_id());

-- ============================================
-- user_profiles: כל אחד רואה פרופילים, רק הבעלים עורך
-- ============================================
CREATE POLICY "Anyone can read profiles"
  ON user_profiles FOR SELECT
  USING (true);

CREATE POLICY "Users can update own profile"
  ON user_profiles FOR UPDATE
  USING (user_id = auth.clerk_user_id());

CREATE POLICY "Users can insert own profile"
  ON user_profiles FOR INSERT
  WITH CHECK (user_id = auth.clerk_user_id());

-- ============================================
-- feedback: כל אחד כותב, רק admin קורא
-- ============================================
CREATE POLICY "Anyone can submit feedback"
  ON feedback FOR INSERT
  WITH CHECK (true);

CREATE POLICY "Users can read own feedback"
  ON feedback FOR SELECT
  USING (user_id = auth.clerk_user_id());

-- ============================================
-- achievements: קריאה בלבד לכולם
-- ============================================
CREATE POLICY "Anyone can read achievements"
  ON achievements FOR SELECT
  USING (true);

-- user_achievements: רק הבעלים
CREATE POLICY "Users can read own achievements"
  ON user_achievements FOR SELECT
  USING (user_id = auth.clerk_user_id());

CREATE POLICY "Users can earn achievements"
  ON user_achievements FOR INSERT
  WITH CHECK (user_id = auth.clerk_user_id());

אימות - בדיקה שהכל עובד

רשימת בדיקות:

  1. התחבר לאפליקציה עם Clerk
  2. לחץ על כפתור "Test RLS Connection" (מהקומפוננטה בשלב 3)
  3. ודא שאתה רואה את ה-user ID בקונסולה
  4. נסה ליצור ספר חדש - ודא שה-user_id נשמר נכון
  5. פתח Incognito ונסה לגשת לספר פרטי - צריך לקבל שגיאה
  6. ודא שספרים ציבוריים נגישים ללא התחברות

🚨 שים לב - Clerk JWT Claims

ודא שה-JWT Template ב-Clerk כולל את השדות הבאים בתוך ה-claims:

{
  "sub": "{{user.id}}",
  "email": "{{user.primary_email_address}}",
  "role": "authenticated",
  "aud": "authenticated",
  "iss": "https://YOUR_CLERK_DOMAIN"
}

השדה sub הוא הקריטי ביותר - הוא מה שה-policies משתמשות בו לזיהוי המשתמש.

משוב שנאסף ממשתמשים ונבדק בסשן הקודם

📋 נקודות עיקריות שעלו

🌐 1. RTL עדיין שבור במקומות רבים

טקסטים בעברית לא מיושרים נכון, כפתורים הפוכים, Layout לא מתאים לכיוון ימין-לשמאל באזורים מסוימים.

🔴 דחוף

📝 2. Auto-fill לא עובד בתצוגה מקדימה

כשבוחרים דמות/סגנון, השדות לא מתמלאים אוטומטית בצפייה מקדימה של הספר.

🔴 דחוף

✨ 3. "הפתע אותי" קיים רק במקום אחד

כפתור "Surprise me" / "הפתע אותי" שמייצר בחירות רנדומליות קיים רק בשלב אחד של יצירת הספר. המשתמשים רוצים אותו בכל שלב.

🟡 שיפור

⚠ 4. שגיאות 404 ב-API

חלק מהקריאות ל-API מחזירות 404. זה קשור למעבר מ-Base44 ל-Supabase ונמצא בטיפול.

🔵 בטיפול

🔤 5. אין ניקוד בטקסטים

לספרי ילדים צעירים (גילאי 4-7) חשוב שיהיה ניקוד. כרגע הטקסטים שה-AI מייצר הם ללא ניקוד.

🟡 שיפור

🖼 6. צריך תמונות Gemini על כרטיסים במקום אימוג'ים

הכרטיסים (דמויות, ספרים, קטגוריות) משתמשים באימוג'ים כאייקונים. עדיף להשתמש באיורים שנוצרו עם Gemini לחוויה עשירה יותר.

🟡 שיפור

🟡

הוספת כפתור "הפתע אותי" בכל שלב של יצירת הספר: בחירת דמות, בחירת סגנון, בחירת ז'אנר, בחירת שפה.
🔵
תיקון שגיאות 404
כבר בטיפול - חלק מהמעבר ל-Supabase. לוודא שכל ה-endpoints מעודכנים.
🟡
הוספת ניקוד לטקסטים
